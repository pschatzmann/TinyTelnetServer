756  static String resolveName(const char* path) {
757  if (String(path).startsWith("/")) {
758  return path;
759  }
760  if (String(path) == ".") return current_dir;
761  if (String(path).startsWith("./")) {
762  path += 2; // Skip "./"
763  return current_dir + (current_dir.endsWith("/") ? "" : "/") + path;
764  }
765  if (String(path).startsWith("../")) {
766  String parent = dirUp(current_dir.c_str());
767  path += 3; // Skip "../"
768  return parent + (parent.endsWith("/") ? "" : "/") + path;
769  }
770  if (String(path) == "..") return dirUp(current_dir.c_str());
771  return current_dir + (current_dir.endsWith("/") ? "" : "/") + path;
