189  int parseTelnetCommands(char* cmds, int len, Client& client) {
190  TELNET_LOGD("parseTelnetCommands: %d", len);
191  int start = 0;
192  char cmd[80];

206  processTelnetCommand(cmd, len, client);
207  start += len;
208  }
209 
210  return start;
211  }

238  void processTelnetCommand(char* cmd, int len, Client& client) {
239  assert(cmd[0] == IAC);
240  TELNET_LOGD("telnet cmd:%s %d (len=%d)", controlStr(cmd[1]), cmd[2], len);
241  // confirm all do request
242  if (cmd[1] == DO) {
243  // DO -> WILL or WONT
244  cmd[1] = WILL;
245  if (cmd[2] == STATUS) {
246  // suppress go ahead
247  cmd[1] = WONT;
248  }
249  client.write(cmd, len);
250  TELNET_LOGD("-> reply:%s %d", controlStr(cmd[1]), cmd[2]);
251  } else if (cmd[1] == WILL) {
252  // WILL -> DO or DONT
253  cmd[1] = DONT;
254  if (cmd[2] == SUPPRESS_GA || cmd[2] == LINEMODE) {
255  // accept line mode negotiation
256  cmd[1] = DO;
257  }
258  client.write(cmd, len);
259  TELNET_LOGD("-> reply:%s %d", controlStr(cmd[1]), cmd[2]);
260  } else if (cmd[1] == SB && cmd[2] == LINEMODE) {
261  // Acknowledges only MODE_EDIT accepted
262  char tmp[7] = {IAC, SB, 34, 1, 01, IAC, SE};
263  TELNET_LOGD("-> reply %d (len=%d)", tmp[2], sizeof(tmp));
264  client.write(tmp, sizeof(tmp));
265  client.println("> Welcome to TinyTelnetServer");
266  }

294  telnet::Vector<telnet::Str>& parameters) {

298  int end;
299  // check if function or list syntax
300  int pos = cmd.indexOf('(');
301  if (pos != -1) {

304  end = cmd.indexOf(')');
305  if (end == -1) {

318  assert(tail.isEmpty());
