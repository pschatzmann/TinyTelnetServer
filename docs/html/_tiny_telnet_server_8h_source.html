149  int parseTelnetCommands(char* cmds, int len, Print& client) {
150  TELNET_LOGD("parseTelnetCommands: %d", len);
151  int start = 0;
152  char cmd[80];
153  while (cmds[start] == IAC) {
154  int len = 3;
155  if (cmds[start + 1] == SB) {
156  // subnegotiation
157  TELNET_LOGD("---> subnegotiation %d", start + 1);
158  int end = start + 2;
159  while (cmds[end] != SE && end < len) {
160  end++;
161  }
162  assert(cmds[end] == SE);
163  len = end - start;
164  }
165  strncpy(cmd, cmds + start, len);
166  processTelnetCommand(cmd, len, client);
167  start += len;
168  }
169 
170  return start;
171  }

198  void processTelnetCommand(char* cmd, int len, Print& client) {
199  assert(cmd[0] == IAC);
200  TELNET_LOGD("telnet cmd:%s %d (len=%d)", controlStr(cmd[1]), cmd[2], len);
201  // confirm all do request
202  if (cmd[1] == DO) {
203  // DO -> WILL or WONT
204  cmd[1] = WILL;
205  if (cmd[2] == STATUS) {
206  // suppress go ahead
207  cmd[1] = WONT;
208  }
209  client.write(cmd, len);
210  TELNET_LOGD("-> reply:%s %d", controlStr(cmd[1]), cmd[2]);
211  } else if (cmd[1] == WILL) {
212  // WILL -> DO or DONT
213  cmd[1] = DONT;
214  if (cmd[2] == SUPPRESS_GA || cmd[2] == LINEMODE) {
215  // accept line mode negotiation
216  cmd[1] = DO;
217  }
218  client.write(cmd, len);
219  TELNET_LOGD("-> reply:%s %d", controlStr(cmd[1]), cmd[2]);
220  } else if (cmd[1] == SB && cmd[2] == LINEMODE) {
221  // Acknowledges only MODE_EDIT accepted
222  char tmp[7] = {IAC, SB, 34, 1, 01, IAC, SE};
223  TELNET_LOGD("-> reply %d (len=%d)", tmp[2], sizeof(tmp));
224  client.write(tmp, sizeof(tmp));
225  client.println("> Welcome to TinyTelnetServer");
226  }
227  }
