143  cmd = input;
144  char delimiter;

146  int end;
147  // check if function or list syntax
148  int pos = cmd.indexOf('(');
149  if (pos != -1) {
150  delimiter = ',';
151  offset = 1;
152  end = cmd.indexOf(')');
153  if (end == -1) {
154  return false;
155  }
156  } else {
157  pos = cmd.indexOf(' ');
158  delimiter = ' ';
159  offset = 0;
160  end = cmd.length();
161  }
162 
163  // determine cmd
164  cmd.substr(input, 0, pos);
165  telnet::Str tail = "";
166  assert(tail.isEmpty());
167  // determine parameters
168  if (pos > 0) tail.substr(input, pos + offset, end);
169  telnet::Str par = "";
