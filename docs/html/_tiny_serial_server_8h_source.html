168  bool parseCommand(const char* input, telnet::Str& cmd,
169  telnet::Vector<telnet::Str>& parameters) {
170  cmd = input;
171  char delimiter;
172  int offset;
173  int end;
174  // check if function or list syntax
175  int pos = cmd.indexOf('(');
176  if (pos != -1) {
177  delimiter = ',';
178  offset = 1;
179  end = cmd.indexOf(')');
180  if (end == -1) {
181  return false;
182  }
183  } else {
184  pos = cmd.indexOf(' ');
185  delimiter = ' ';
186  offset = 0;
187  end = cmd.length();
188  }
189 
190  // determine cmd
191  cmd.substr(input, 0, pos);
192  telnet::Str tail = "";
193  assert(tail.isEmpty());
194  // determine parameters
195  if (pos > 0) tail.substr(input, pos + offset, end);
196  telnet::Str par = "";
197  TELNET_LOGI("cmd: '%s'", cmd.c_str());
198 
199  while (!tail.isEmpty()) {
200  tail.trim();
201  split(tail, par, tail, delimiter);
203  if (is_remove_quotes && par.startsWith("\"") && par.endsWith("\"")) {
204  par.substr(par, 1, par.length());
205  par.trim();
206  }
207  TELNET_LOGI("- par: '%s'", par.c_str());
208  parameters.push_back(par);
209  }
210  return true;
